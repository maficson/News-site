from bs4 import BeautifulSoup
import random
from googlesearch import search  # Потрібно встановити googlesearch-python через pip

async def find_word_in_page(session, url, word):
    headers = {
        "User-Agent": "Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/119.0.0.0 Safari/537.36"
    }
    try:
        async with session.get(url, headers=headers, timeout=aiohttp.ClientTimeout(total=10)) as response:
            if response.status != 200:
                return None

            page_content = await response.text()
            soup = BeautifulSoup(page_content, 'html.parser')

            if word.lower() in soup.get_text().lower():
                return url
            return None
    except Exception as e:
        print(f"Error on {url}: {e}")
        return None

async def search_word_in_sites(word, urls):
    async with aiohttp.ClientSession() as session:
        tasks = []
        for url in urls:
            tasks.append(find_word_in_page(session, url, word))

        results = await asyncio.gather(*tasks)
        return [url for url in results if url]

def generate_random_sites_from_google(query, num_sites=200):
    return list(search(query, num_results=num_sites))

def generate_top_sites(num_sites=100):
    top_sites = [
        "https://google.com", "https://youtube.com", "https://facebook.com", "https://baidu.com", "https://wikipedia.org",
        "https://amazon.com", "https://twitter.com", "https://instagram.com", "https://yahoo.com", "https://reddit.com",
        "https://linkedin.com", "https://live.com", "https://vk.com", "https://tumblr.com", "https://pinterest.com",
        "https://wordpress.com", "https://microsoft.com", "https://netflix.com", "https://ebay.com", "https://t.co",
        "https://apple.com", "https://github.com", "https://cnn.com", "https://bbc.com", "https://quora.com",
        "https://wordpress.org", "https://myspace.com", "https://yandex.ru", "https://telegram.org", "https://stackoverflow.com",
        "https://forbes.com", "https://bbc.co.uk", "https://amazon.co.uk", "https://merriam-webster.com", "https://bbc.ca"
    ]
    return random.sample(top_sites, num_sites)

search_word = input("Введите слово для поиска на сайтах (например Python): ")
print("Выберите метод поиска:")
print("1. Поиск через Google")
print("2. Поиск через топ-100 популярных сайтов")
method = input("Ваш выбор (1 или 2): ")

if method == '1':
    query = search_word
    urls_to_search = generate_random_sites_from_google(query, 100)
elif method == '2':
    urls_to_search = generate_top_sites(100)
else:
    print("Неверный выбор!")
    exit()

found_links = asyncio.run(search_word_in_sites(search_word, urls_to_search))

if found_links:
    print("Сайты, где найдено слово:")
    for link in found_links:
        print(link)
else:
    print(f"Слово '{search_word}' не найдено на указанных сайтах.")
